  <script>
    function fall() {
      const apple = document.getElementById('apple');
      apple.style.transition = "transform 1s ease-in";
      apple.style.transform = "translateY(100vh)";
      setTimeout(() => {
        document.body.style.overflowY = "auto";
        document.getElementById('main').style.opacity = "1";
        window.scrollTo({ top: window.innerHeight, behavior: 'smooth' });
      }, 1000);
    }

    function copyMessage() {
      const text = document.getElementById('msg').value.trim();
      if (!text) return alert('Введите сообщение перед копированием.');
      navigator.clipboard.writeText(text)
        .then(() => alert('Сообщение скопировано в буфер.'))
        .catch(() => alert('Не удалось скопировать автоматически.'));
    }

    function togglePhone() {
      const ph = document.getElementById('phone');
      ph.style.display = (ph.style.display = 'none' || ph.style.display = '') ? 'inline' : 'none';
    }
  </script>
